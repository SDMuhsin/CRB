# Tuned for Alliance Canada wheelhouse compatibility (Nibi, Python 3.11).
#
# Strategy:
#   - Lower bounds capture the API requirements (Qwen2.5 needs transformers
#     >= 4.39; numpy 2.x breaks torch + numba builds; etc.).
#   - NO upper bounds (except numpy<2) so `pip install --no-index` is free
#     to resolve whatever version Alliance currently ships, instead of
#     forcing a PyPI fallback that conflicts with `+computecanada` wheels.
#
# Install on a Nibi compute env:
#     module load gcc arrow scipy-stack cuda cudnn python/3.11
#     virtualenv --no-download env && source env/bin/activate
#     pip install --no-index --upgrade pip
#     pip install --no-index -r requirements.txt
#
# If --no-index reports a missing wheel for ONE package (most likely
# flash1dkmeans), drop --no-index for that single install:
#     pip install flash1dkmeans

# ---- Core PTQ / eval stack (all in Alliance wheelhouse) ----------------------
torch>=2.3                      # 2.3.0+cc and 2.5.1+cc both validated
transformers>=4.42              # Qwen2.5 needs >=4.39; CC stocks 4.42.x
tokenizers>=0.19                # transformers 4.42 uses tokenizers 0.19.x
huggingface_hub>=0.23           # transformers 4.42 + datasets 2.20 minimum
safetensors>=0.4
accelerate>=0.30
datasets>=2.18                  # CC stocks 2.20.0
pyarrow>=12                     # provided by `module load arrow`
sentencepiece>=0.2
numpy>=1.24,<2                  # CC has 1.26.4+cc; numpy 2.x is incompatible
exceptiongroup

# ---- GuidedQuant / LNQ (src/run_lnq.py only) --------------------------------
# numba is in the Alliance wheelhouse. flash1dkmeans is a small pure-numba
# helper that is *not* in the wheelhouse — install separately from PyPI:
#     pip install flash1dkmeans
numba>=0.58
flash1dkmeans>=0.1

# ---- Utilities --------------------------------------------------------------
pyparsing>=3.0                  # utils/autosearch.py
tqdm>=4.60
